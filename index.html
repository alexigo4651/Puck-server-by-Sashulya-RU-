🚀 Инфраструктура серверов Puck

Установка и настройка сервера (Обновлено для Build 202)

Требования:
  • Сервер на базе Debian (Debian 12 bookworm) с SSH-доступом.
  • Root-доступ (все команды выполняются от имени пользователя root).
  • Минимум 2 ГБ ОЗУ (каждый сервер Puck потребляет ~300 МБ).
  • Базовые знания командной строки.
Примечание: Для работы сервера будет создан отдельный пользователь `puckd`.

Шаг 1: Обновление системы

apt update && apt upgrade -y

Шаг 2: Установка зависимостей и SteamCMD

Установите необходимые инструменты и подготовьте SteamCMD:

apt install software-properties-common
echo "deb http://deb.debian.org/debian bookworm main contrib non-free" > /etc/apt/sources.list.d/nonfree.list
dpkg --add-architecture i386
apt update
apt install steamcmd

Шаг 3: Создание папки для загрузки сервера

mkdir -p /srv/puck-download

Шаг 4: Загрузка файлов сервера Puck

Используйте SteamCMD для загрузки файлов выделенного сервера (App ID 3481440):

/usr/games/steamcmd +force_install_dir /srv/puck-download +login anonymous +app_update 3481440 +quit

Шаг 5: Создание системного пользователя

Создайте выделенного пользователя для безопасного запуска сервера:

adduser --system --no-create-home --group puckd

Шаг 6: Настройка первого экземпляра сервера

Создайте директорию для первого сервера и назначьте права:

mkdir -p /srv/puck/server1
cp -r /srv/puck-download/* /srv/puck/server1/
chown -R puckd:puckd /srv/puck/server1
chmod +x /srv/puck/server1/start_server.sh
chmod +x /srv/puck/server1/Puck.x86_64

Шаг 7: Создание конфигурации сервера

Откройте файл конфигурации:

nano /srv/puck/server1/config.json

Вставьте и измените по необходимости следующий код:

{
  "port": 7777,
  "pingPort": 7778,
  "name": "My Dedicated Puck Server",
  "maxPlayers": 10,
  "password": "",
  "voip": false,
  "isPublic": true,
  "adminSteamIds": [],
  "reloadBannedSteamIds": false,
  "usePuckBannedSteamIds": true,
  "printMetrics": true,
  "kickTimeout": 300,
  "sleepTimeout": 60,
  "joinMidMatchDelay": 10,
  "targetFrameRate": 120,
  "serverTickRate": 100,
  "clientTickRate": 200,
  "startPaused": false,
  "allowVoting": true,
  "phaseDurationMap": {
    "Warmup": 600,
    "FaceOff": 3,
    "Playing": 300,
    "BlueScore": 5,
    "RedScore": 5,
    "Replay": 10,
    "PeriodOver": 15,
    "GameOver": 15
  },
  "mods": [
    {
      "id": 3493628417,
      "enabled": false,
      "clientRequired": true
    },
    {
      "id": 3491056040,
      "enabled": true,
      "clientRequired": false
    }
  ]
}
Важные настройки:
• Name: Имя сервера в браузере. Поддерживает Unity Rich Text (цвета и размеры), но не злоупотребляйте этим.
• Port: Порт сервера (7777 по умолчанию). Его нужно пробросить в роутере или включить UPnP.
• targetFrameRate: Фактический FPS сервера (рекомендуется ставить на 10-20 выше тикрейта).
• serverTickRate: Частота обновления сервера. Рекомендуется 240-360Hz+, если позволяет железо.
• clientTickRate: Частота отправки данных от клиентов (влияет на инпут-лаг). Можно ставить равной серверному тикрейту.
• Mods: В примере добавлены Toaster's Crispy Shadows (клиентский) и GAFURIX' Custom MOTD (серверный).

Шаг 8: Создание службы systemd

Создайте юнит для удобного управления сервером:

nano /etc/systemd/system/puck@.service

Вставьте следующее содержимое:

[Unit]
Description=Puck Dedicated Server - %i
After=network.target

[Service]
WorkingDirectory=/srv/puck/%i
User=puckd
ExecStart=/srv/puck/%i/start_server.sh --serverConfigurationPath /srv/puck/%i/config.json
Restart=on-failure

[Install]
WantedBy=multi-user.target

2. Скрипт объединения баз (merge_steams.sh)

cat << 'EOF' > /root/merge_steams.sh
#!/bin/bash
MASTER_DB="/srv/puck/server1/Logs/steam_db.csv"
INCOMING_DIR="/tmp/incoming_steam_data"
TMP_MERGE="/tmp/merged_db.csv"

touch "$MASTER_DB"

cat "$MASTER_DB" "$INCOMING_DIR"/*.csv 2>/dev/null | awk -F';' '
{
    sid=$1; ips=$2; nicks=$3
    if (sid == "" || sid == "SteamID") next

    split(ips, ip_arr, ","); for (i in ip_arr) if (ip_arr[i] != "" && ip_arr[i] != "n/a") all_ips[sid][ip_arr[i]] = 1
    split(nicks, nick_arr, ","); for (i in nick_arr) if (nick_arr[i] != "" && nick_arr[i] != "n/a") all_nicks[sid][nick_arr[i]] = 1
}
END {
    for (sid in all_ips) {
        ip_str = ""; for (ip in all_ips[sid]) { ip_str = ip_str ? ip_str "," ip : ip }
        if (ip_str == "") ip_str = "n/a"

        nick_str = ""; for (nick in all_nicks[sid]) { nick_str = nick_str ? nick_str "," nick : nick }
        if (nick_str == "") nick_str = "n/a"

        print sid ";" ip_str ";" nick_str
    }
}' > "$TMP_MERGE"

mv "$TMP_MERGE" "$MASTER_DB"
EOF
chmod +x /root/merge_steams.sh

2. Парсер логов (update_steam_db.sh)

/srv/puck/server1/Logs/update_steam_db.sh 
cat << 'EOF' > /srv/puck/server1/Logs/update_steam_db.sh
#!/bin/bash

# Пути к файлам
LOG_DIR="/srv/puck/server1/Logs"
DB_FILE="$LOG_DIR/steam_db.csv"
TMP_IPS="/tmp/steam_ips.txt"
TMP_NICKS="/tmp/steam_nicks.txt"

# Создаем директорию и пустую базу, если их нет
mkdir -p "$LOG_DIR"
if [ ! -f "$DB_FILE" ]; then
    echo "SteamID;IP_Addresses;Nicknames" > "$DB_FILE"
fi

# 1. Сбор IP из файлов .ndjson (используем jq)
cat "$LOG_DIR"/ip_logger_*.ndjson 2>/dev/null | \
jq -R -r 'fromjson? | select(.steam_id != null and .ip != null) | "\(.steam_id)\t\(.ip)"' | \
awk '{
    split(a[$1], ips, ",")
    found = 0
    for (j in ips) if (ips[j] == $2) found = 1
    if (!found) a[$1] = a[$1] ? a[$1] "," $2 : $2
} END { for (k in a) print k "\t" a[k] }' > "$TMP_IPS"

# 2. Сбор ников из Puck*.log (используем gawk)
gawk '
{
    sid=""; nick=""
    if (match($0, /Steam Id ([0-9]{17}).*\(([^)]+)\)/, m)) {
        sid = m[1]; nick = m[2]
    } 
    else if (match($0, /\[UIChat\] \[[^\]]+\] (.*) \([0-9]+\) \[([0-9]{17})\]:/, c)) {
        nick = c[1]; sid = c[2]
    }
    if (sid != "" && nick != "") {
        gsub(/^[ \t]+|[ \t]+$/, "", nick)
        if (nick ~ /^[0-9]{17}$/) next
        split(nicks[sid], ns, ",")
        found = 0
        for (i in ns) if (ns[i] == nick) found = 1
        if (!found) nicks[sid] = nicks[sid] ? nicks[sid] "," nick : nick
    }
} END { for (k in nicks) print k "\t" nicks[k] }' "$LOG_DIR"/Puck*.log 2>/dev/null > "$TMP_NICKS"

# 3. Слияние данных
awk '
BEGIN { FS="[;\t]"; OFS=";" }
function merge(old, new,    arr, k, res, uniq) {
    split(old "," new, arr, ",")
    for (k in arr) if (arr[k] != "" && arr[k] != "n/a") uniq[arr[k]] = 1
    res = ""
    for (k in uniq) res = (res == "") ? k : res "," k
    return (res == "" ? "n/a" : res)
}
{
    if ($1 == "SteamID") next
    sid = $1
    if (FILENAME == db) { i[sid]=$2; n[sid]=$3 }
    else if (FILENAME == t_i) i[sid] = merge(i[sid], $2)
    else if (FILENAME == t_n) n[sid] = merge(n[sid], $2)
}
END {
    print "SteamID;IP_Addresses;Nicknames"
    for (sid in i) sids[sid]=1; for (sid in n) sids[sid]=1
    for (sid in sids) {
        print sid, (i[sid]?i[sid]:"n/a"), (n[sid]?n[sid]:"n/a")
    }
}' db="$DB_FILE" t_i="$TMP_IPS" t_n="$TMP_NICKS" "$DB_FILE" "$TMP_IPS" "$TMP_NICKS" > "${DB_FILE}.tmp"

mv -f "${DB_FILE}.tmp" "$DB_FILE"
rm -f "$TMP_IPS" "$TMP_NICKS"
EOF
chmod +x /srv/puck/server1/Logs/update_steam_db.sh

3. Просмотр Базы (view_db.sh)

/srv/puck/server1/Logs/view_db.sh 
cat << 'EOF' > /srv/puck/server1/Logs/view_db.sh
#!/bin/bash
DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
DB_FILE="$DIR/steam_db.csv"
BAN_FILE="/srv/puck/server1/manual_bans.txt"

[ ! -f "$DB_FILE" ] && { echo "Файл базы не найден!"; exit 1; }

# Цвета
YELLOW='\033[1;33m'; RESET='\033[0m'; GRAY='\033[0;90m'; BLUE='\033[0;34m'
CYAN='\033[0;36m'; RED='\033[0;31m'

W_SID=20; W_IP=40; W_NICK=50

# 1. Подготовка данных (пропускаем заголовок файла через NR>1)
temp_sorted=$(awk -F';' -v ban_file="$BAN_FILE" '
BEGIN {
    if ((getline < ban_file) > 0) {
        do { if ($0 != "") bans[$0] = 1 } while ((getline < ban_file) > 0)
    }
    close(ban_file)
}
# Пропускаем пустые строки и заголовок самого CSV
NR>1 && $1 != "" && $1 != "SteamID" {
    sid=$1; ips=$2; nicks=$3
    is_banned = 0
    split(ips, ip_arr, ",")
    for (i in ip_arr) { if (ip_arr[i] in bans) { is_banned = 1; break } }

    if (is_banned) { prio = 3 }
    else if (nicks == "" || nicks == "n/a" || nicks ~ /^[[:space:]]*$/) { prio = 2 }
    else { prio = 1 }

    print prio ";" sid ";" ips ";" nicks
}' "$DB_FILE" | sort -t';' -k1,1n -k2,2n)

# 2. Отрисовка
echo "$temp_sorted" | awk -F';' -v Y="$YELLOW" -v R="$RESET" -v G="$GRAY" -v B="$BLUE" -v C="$CYAN" -v RED="$RED" \
    -v w1="$W_SID" -v w2="$W_IP" -v w3="$W_NICK" '
function draw_line() {
    printf G "+"; for(i=0;i 0) {
        draw_line()
        printf "Всего: " Y "%d" R " | Забанено: " RED "%d" R " | Чисто: " C "%d" R "\n", total, banned_count, total-banned_count
        printf "С никами: " G "%d" R " | Без ников: " G "%d" R "\n", with_nick, no_nick
    } else {
        print "База данных пуста."
    }
}'
EOF
chmod +x /srv/puck/server1/Logs/view_db.sh

4. Мониторинг входов (live_logs.sh)

/srv/puck/server1/Logs/live_logs.sh 
cat << 'EOF' > /srv/puck/server1/Logs/live_logs.sh
#!/bin/bash
LOG_DIR="/srv/puck/server1/Logs"
DB_FILE="$LOG_DIR/steam_db.csv"
LATEST_LOG=$(ls -1t $LOG_DIR/ip_logger_*.ndjson 2>/dev/null | head -n 1)
[ -z "$LATEST_LOG" ] && { echo "Логи не найдены!"; exit 1; }

Y="\033[1;33m"; G="\033[1;32m"; R="\033[1;31m"; RST="\033[0m"; GRY="\033[0;90m"; CYN="\033[1;36m"
echo -e "${GRY}Загрузка последних входов из $LATEST_LOG...${RST}"
printf "${Y}%-20s | %-24s | %-15s | %-10s | %s${RST}\n" "SteamID" "Time (UTC)" "IP" "Status" "Nickname"
echo -e "${GRY}-------------------------------------------------------------------------------------------------------${RST}"

grep "APPROVED" "$LATEST_LOG" | tail -n 40 | jq -r 'select(.steam_id != null) | "\(.steam_id) \(.ts_utc) \(.ip) \(.decision)"' | awk -v db="$DB_FILE" '
BEGIN { while ((getline < db) > 0) { split($0, row, ";"); s=row[1]; gsub(/[ \t]+/, "", s); n[s]=row[3] }; close(db); Y="\033[1;33m"; G="\033[1;32m"; RST="\033[0m" }
{ sid=$1; ts=$2; ip=$3; dec=$4; name = (sid in n) ? Y n[sid] RST : "\033[1;30mNew Player\033[0m"; printf "%-20s | %-24s | %-15s | %s%-10s%s | %s\n", sid, ts, ip, G, dec, RST, name }'

echo -e "${CYN}>>> РЕЖИМ LIVE ЗАПУЩЕН (мониторинг событий)${RST}"
tail -q -n 0 -F "$LATEST_LOG" 2>/dev/null | jq --unbuffered -R -r 'fromjson? | select(.steam_id != null) | "\(.steam_id) \(.ts_utc) \(.ip) \(.decision)"' | awk -v db="$DB_FILE" '
BEGIN { while ((getline < db) > 0) { split($0, row, ";"); s=row[1]; gsub(/[ \t]+/, "", s); n[s]=row[3] }; close(db); Y="\033[1;33m"; G="\033[1;32m"; R="\033[1;31m"; RST="\033[0m" }
{ sid=$1; ts=$2; ip=$3; dec=$4; name = (sid in n) ? Y n[sid] RST : R "New Player" RST; st_fmt = (dec ~ /^[Aa]/) ? G sprintf("%-10s", dec) RST : R sprintf("%-10s", dec) RST; printf "%-20s | %-24s | %-15s | %s | %s\n", sid, ts, ip, st_fmt, name }'
EOF
chmod +x /srv/puck/server1/Logs/live_logs.sh
