cat << 'EOF' > /srv/puck/server1/Logs/update_steam_db.sh #!/bin/bash # Пути к файлам (проверь, что они совпадают с твоими) LOG_DIR="/srv/puck/server1/Logs" DB_FILE="$LOG_DIR/steam_db.csv" TMP_IPS="/tmp/steam_ips.txt" TMP_NICKS="/tmp/steam_nicks.txt" # Создаем пустую базу с заголовками, если её нет if [ ! -f "$DB_FILE" ]; then echo "SteamID;IP_Addresses;Nicknames" > "$DB_FILE" fi echo "Начинаю сбор данных из логов..." # 1. Сбор IP из файлов .ndjson (используем jq с защитой от битых строк) # Вытаскиваем SteamID и IP, разделяем табуляцией для awk cat "$LOG_DIR"/ip_logger_*.ndjson 2>/dev/null | \ jq -R -r 'fromjson? | select(.steam_id != null and .ip != null) | "\(.steam_id)\t\(.ip)"' | \ awk '{ split(a[$1], ips, ",") found = 0 for (j in ips) if (ips[j] == $2) found = 1 if (!found) a[$1] = a[$1] ? a[$1] "," $2 : $2 } END { for (k in a) print k "\t" a[k] }' > "$TMP_IPS" # 2. Сбор ников из Puck*.log (используем gawk для корректной работы match) gawk ' { sid=""; nick="" # Вариант 1: Поиск в системных сообщениях (учитываем любые слова между ID и ником) if (match($0, /Steam Id ([0-9]{17}).*\(([^)]+)\)/, m)) { sid = m[1]; nick = m[2] } # Вариант 2: Поиск в сообщениях чата else if (match($0, /\[UIChat\] \[[^\]]+\] (.*) \([0-9]+\) \[([0-9]{17})\]:/, c)) { nick = c[1]; sid = c[2] } if (sid != "" && nick != "") { # Чистим ник от лишних пробелов по краям gsub(/^[ \t]+|[ \t]+$/, "", nick) # Проверка: если в скобках вместо ника попал еще один SteamID, пропускаем if (nick ~ /^[0-9]{17}$/) next split(nicks[sid], ns, ",") found = 0 for (i in ns) if (ns[i] == nick) found = 1 if (!found) nicks[sid] = nicks[sid] ? nicks[sid] "," nick : nick } } END { for (k in nicks) print k "\t" nicks[k] }' "$LOG_DIR"/Puck*.log 2>/dev/null > "$TMP_NICKS" # 3. Слияние новых данных со старой базой # FS (разделитель на входе) - либо точка с запятой, либо табуляция awk ' BEGIN { FS="[;\t]"; OFS=";" } function merge(old, new, arr, k, res, uniq) { # Объединяем старые и новые значения через запятую split(old "," new, arr, ",") for (k in arr) if (arr[k] != "" && arr[k] != "n/a") uniq[arr[k]] = 1 # Собираем уникальный список обратно в строку res = "" for (k in uniq) res = (res == "") ? k : res "," k return (res == "" ? "n/a" : res) } { if ($1 == "SteamID") next sid = $1 if (FILENAME == db) { i[sid]=$2; n[sid]=$3 } else if (FILENAME == t_i) i[sid] = merge(i[sid], $2) else if (FILENAME == t_n) n[sid] = merge(n[sid], $2) } END { print "SteamID;IP_Addresses;Nicknames" # Собираем все SteamID, которые встретились в любом из файлов for (sid in i) sids[sid]=1; for (sid in n) sids[sid]=1 for (sid in sids) { print sid, (i[sid]?i[sid]:"n/a"), (n[sid]?n[sid]:"n/a") } }' db="$DB_FILE" t_i="$TMP_IPS" t_n="$TMP_NICKS" "$DB_FILE" "$TMP_IPS" "$TMP_NICKS" > "${DB_FILE}.tmp" # Сохраняем результат и удаляем временные файлы mv "${DB_FILE}.tmp" "$DB_FILE" rm -f "$TMP_IPS" "$TMP_NICKS" echo -e "\n\033[1;32m✅ БАЗА ОБНОВЛЕНА И СИНХРОНИЗИРОВАНА\033[0m" EOF chmod +x /srv/puck/server1/Logs/update_steam_db.sh
